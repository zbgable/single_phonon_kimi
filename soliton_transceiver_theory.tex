\begin{abstract}
This document establishes the complete theory of soliton-based single-phonon quantum transceiver system. The system consists of three parts: (1) superconducting Transmon qubit emits sech-type single-phonon wavepacket through IDT coupler; (2) the wavepacket propagates as soliton in LN phonon waveguide with dispersion, nonlinearity and loss; (3) receiver absorbs single phonon via time-reversal process. We derive soliton conditions, calculate wavepacket length, and compute absorption efficiency.
\end{abstract}

\section{Transmitter: Single-Phonon Emission}
%=============================================================================

\subsection{Qubit-IDT Coupling Hamiltonian}

In the rotating frame, the Hamiltonian for Transmon qubit coupled to IDT:
\begin{equation}
\hat{H} = -\frac{\hbar\alpha}{2}\hat{n}(\hat{n}-1) + \hbar\Omega(t)\hat{\sigma}_x/2
\end{equation}

The drive field $\Omega(t)$ is designed as sech shape to generate soliton phonon wavepacket:
\begin{equation}
\Omega(t) = \Omega_0\text{sech}\left(\frac{t-t_0}{\tau_0}\right)
\end{equation}

\subsection{Lindblad Master Equation}

Density matrix evolution:
\begin{equation}
\frac{d\hat{\rho}}{dt} = -\frac{i}{\hbar}[\hat{H},\hat{\rho}] + \mathcal{L}_{\gamma_1}[\hat{\rho}] + \mathcal{L}_{\gamma_\phi}[\hat{\rho}] + \mathcal{L}_{\kappa}[\hat{\rho}]
\end{equation}

where IDT emission term:
\begin{equation}
\mathcal{L}_{\kappa}[\hat{\rho}] = \kappa_{\text{IDT}}\left(\hat{\sigma}_-\hat{\rho}\hat{\sigma}_+ - \frac{1}{2}\{\hat{\sigma}_+\hat{\sigma}_-,\hat{\rho}\}\right)
\end{equation}

\subsection{Emitted Phonon Wavepacket}

Number of emitted phonons:
\begin{equation}
N_{\text{ph}} = \int_{-\infty}^{\infty}\kappa_{\text{IDT}}P_e(t)dt
\end{equation}

Emitted wavepacket shape:
\begin{equation}
u(0,t) = \sqrt{N_{\text{ph}}}\text{sech}\left(\frac{t}{\tau_0}\right)
\end{equation}

\section{Propagation: Phonon Soliton in LN Waveguide}
%=============================================================================

\subsection{Nonlinear Schrodinger Equation}

The phonon wavepacket $u(z,t)$ propagation in LN waveguide satisfies NLSE:
\begin{equation}
i\frac{\partial u}{\partial z} = -\frac{\beta_2}{2}\frac{\partial^2 u}{\partial t^2} + \gamma|u|^2u - i\frac{\alpha}{2}u
\label{eq:NLSE_loss}
\end{equation}

Physical meaning of terms:
\begin{itemize}
\item $-\frac{\beta_2}{2}\partial_t^2u$: Group Velocity Dispersion (GVD)
\item $\gamma|u|^2u$: Kerr nonlinearity (phonon-phonon interaction)
\item $-i\frac{\alpha}{2}u$: Propagation loss
\end{itemize}

\subsection{Soliton Solution Derivation}

\begin{derivation}[Fundamental Soliton]
\label{deriv:soliton}

Neglecting loss ($\alpha=0$), we try ansatz:
\begin{equation}
u(z,t) = \sqrt{P_0}\text{sech}\left(\frac{t}{\tau_0}\right)e^{i\phi(z)}
\end{equation}

Substituting into NLSE and separating real and imaginary parts:

1. \textbf{Phase evolution}:
\begin{equation}
\frac{d\phi}{dz} = \frac{\gamma P_0}{2} = \text{const}
\end{equation}

2. \textbf{Amplitude condition} (Key!):

Dispersion term:
\begin{equation}
\frac{\partial^2 u}{\partial t^2} = \frac{\sqrt{P_0}}{\tau_0^2}\left[\text{sech}(x) - 2\text{sech}^3(x)\right]e^{i\phi}
\end{equation}
where $x = t/\tau_0$.

Nonlinear term:
\begin{equation}
\gamma|u|^2u = \gamma P_0\sqrt{P_0}\text{sech}^3(x)e^{i\phi}
\end{equation}

Balance condition requires equal coefficients for $\text{sech}^3(x)$:
\begin{equation}
-\frac{\beta_2}{2}\left(-\frac{2}{\tau_0^2}\right) = \gamma P_0
\end{equation}

Resulting in \textbf{soliton condition}:
\begin{equation}
\boxed{\frac{\gamma P_0\tau_0^2}{|\beta_2|} = 1}
\label{eq:soliton_condition}
\end{equation}
\end{derivation}

\subsection{Soliton Parameter Calculation}

\begin{theorem}[Fundamental Soliton Parameters]
Given pulse width $\tau_0$ and waveguide parameters $\beta_2,\gamma$, the fundamental soliton ($N=1$) parameters are:
\begin{align}
P_0 &= \frac{|\beta_2|}{\gamma\tau_0^2} \\
A_0 &= \sqrt{P_0} = \sqrt{\frac{|\beta_2|}{\gamma}}\frac{1}{\tau_0}
\end{align}
\end{theorem}

\textbf{Numerical calculation} (with experimental parameters):
\begin{align}
P_0 &= \frac{10^{-20}}{10^{-3}\times(10^{-9})^2} = 10^{-2}\text{ W} = 10\text{ mW} \\
A_0 &= \sqrt{10^{-2}} = 0.1\text{ W}^{1/2}
\end{align}

\subsection{Wavepacket Length}

FWHM of sech wavepacket:
\begin{equation}
\Delta t_{\text{FWHM}} = 2\tau_0\text{arccosh}(\sqrt{2}) \approx 1.76\tau_0
\end{equation}

Corresponding \textbf{spatial length}:
\begin{equation}
\boxed{L_{\text{pulse}} = v_g\Delta t_{\text{FWHM}} \approx 1.76v_g\tau_0}
\end{equation}

For $\tau_0 = 1$ ns, $v_g = 3800$ m/s:
\begin{equation}
L_{\text{pulse}} \approx 1.76\times3800\times10^{-9} \approx 6.7\text{ um}
\end{equation}

\subsection{Characteristic Lengths}

\begin{definition}[Characteristic Lengths]
\begin{itemize}
\item \textbf{Dispersion length}: $L_D = \frac{\tau_0^2}{|\beta_2|}$
\item \textbf{Nonlinear length}: $L_{NL} = \frac{1}{\gamma P_0}$
\item \textbf{Loss length}: $L_{\alpha} = \frac{1}{\alpha}$
\end{itemize}
\end{definition}

For soliton: $L_D = L_{NL}$ (from soliton condition).

Numerical estimates:
\begin{align}
L_D &= \frac{(10^{-9})^2}{10^{-20}} = 100\text{ um} \\
L_{NL} &= \frac{1}{10^{-3}\times10^{-2}} = 100\text{ um} \\
L_{\alpha} &= \frac{1}{0.5} = 2\text{ m} \gg L
\end{align}

Since $L \ll L_{\alpha}$, loss effect is small; $L \sim L_D$, both dispersion and nonlinearity are important.

\section{Receiver: Time-Reversal Absorption}
%=============================================================================

\subsection{Time-Reversal Principle}

Maxwell equations and Schrodinger equation are time-symmetric (ignoring dissipation). For wave propagation:
\begin{equation}
u_{\text{rev}}(t) = u^*(-t)
\end{equation}

Physical process:
\begin{enumerate}
\item Receive wavepacket $u(L,t)$
\item Time reversal: $u_{\text{rev}}(t) = u^*(L,-t)$
\item Emit from receiver (direction reversed)
\item Wavepacket "contracts" and is absorbed by qubit
\end{enumerate}

\subsection{Single-Phonon Absorption Efficiency}

\begin{theorem}[Absorption Efficiency]
Let IDT coupling rate be $\kappa_{\text{ext}}$ and qubit total decay $\kappa_{\text{tot}} = \gamma_1 + \kappa_{\text{ext}}$, then single-phonon absorption probability is:
\begin{equation}
\boxed{P_{\text{abs}} = \frac{4\eta}{(1+\eta)^2}}
\end{equation}
where $\eta = \kappa_{\text{ext}}/\kappa_{\text{tot}}$ is coupling efficiency.
\end{theorem}

\textbf{Maximum efficiency}: When $\eta = 1$ (critical coupling), $P_{\text{abs}} = 1$ (100\%).

\subsection{Total Transmission Efficiency}

Considering all stages:
\begin{equation}
\eta_{\text{total}} = \eta_{\text{emit}}\times\eta_{\text{prop}}\times\eta_{\text{abs}}
\end{equation}

Typical values:
\begin{itemize}
\item Emission efficiency: $\eta_{\text{emit}} \sim 80\%$
\item Propagation efficiency: $\eta_{\text{prop}} = e^{-\alpha L} \sim 95\%$
\item Absorption efficiency: $\eta_{\text{abs}} \sim 80\%$
\item Total efficiency: $\eta_{\text{total}} \sim 60\%$
\end{itemize}

\subsection{Split-Step Fourier Propagation}

\begin{verbatim}
def ssfm_propagate(u0, z, beta2, gamma, alpha):
    u = np.zeros((len(z), len(t)), dtype=complex)
    u[0, :] = u0
    u_current = u0.copy()
    
    for n in range(1, len(z)):
        # Nonlinear + loss (half step)
        u_half = u_current * np.exp(
            (-alpha/2 + 1j*gamma*abs(u_current)**2) * dz/2)
        
        # Dispersion (full step, Fourier domain)
        u_fft = np.fft.fft(u_half)
        u_fft *= exp(1j*beta2/2*omega**2*dz)
        u_disp = np.fft.ifft(u_fft)
        
        # Nonlinear + loss (half step)
        u[n,:] = u_disp * np.exp(
            (-alpha/2 + 1j*gamma*abs(u_disp)**2) * dz/2)
    
    return u
\end{verbatim}

This scheme provides a feasible pathway for quantum state transfer in solid-state quantum networks.
